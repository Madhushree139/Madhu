#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>
#include <Adafruit_INA219.h>
#include <SPI.h>
#include <mcp2515.h>

// ── OLED ──────────────────────────────────────────────────
#define SCREEN_WIDTH  128
#define SCREEN_HEIGHT  64
#define OLED_RESET     -1
Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, OLED_RESET);

// ── INA219 Current/Voltage Sensor ─────────────────────────
Adafruit_INA219 ina219;

// ── CAN Bus ───────────────────────────────────────────────
MCP2515 mcp2515(10);   // CS pin = 10
struct can_frame canMsg;

// ── Pins ──────────────────────────────────────────────────
#define TEMP_PIN        A1   // NTC Thermistor
#define RELAY_PIN        7   // Protection relay (LOW = ON)
#define BUZZER_PIN       6   // Alert buzzer

// ── Battery Thresholds ────────────────────────────────────
#define VOLTAGE_MAX     54.0   // Overvoltage threshold  (V)
#define VOLTAGE_MIN      9.0   // Undervoltage threshold (V)
#define CURRENT_MAX     45.0   // Overcurrent threshold  (A)
#define TEMP_MAX        75.0   // Overtemperature limit  (°C)

// ── Battery Capacity ──────────────────────────────────────
#define BATTERY_CAPACITY_AH  10.0   // Battery capacity in Ah

// ── Global Variables ──────────────────────────────────────
float busVoltage   = 0.0;
float current_A    = 0.0;
float temperature  = 0.0;
float SOC          = 100.0;   // State of Charge (%)
float SOH          = 100.0;   // State of Health (%)
float chargeUsed_Ah = 0.0;
bool  faultActive  = false;

unsigned long lastTime = 0;
unsigned long lastCanSend = 0;

// ── Read Temperature from NTC Thermistor ──────────────────
float readTemperature() {
  int raw = analogRead(TEMP_PIN);
  float resistance = (1023.0 / raw - 1.0) * 10000.0;  // 10kΩ pull-up
  float tempK = 1.0 / (log(resistance / 10000.0) / 3950.0 + 1.0 / 298.15);
  return tempK - 273.15;
}

// ── Estimate SOC using voltage mapping ────────────────────
float estimateSOC(float voltage) {
  if (voltage >= 50.4) return 100.0;
  if (voltage >= 48.0) return map(voltage * 10, 480, 504, 80, 100);
  if (voltage >= 44.4) return map(voltage * 10, 444, 480, 20, 80);
  if (voltage >= 42.0) return map(voltage * 10, 420, 444, 5, 20);
  return 0.0;
}

// ── Check protection conditions ───────────────────────────
void checkProtection() {
  faultActive = false;
  String faultMsg = "";

  if (busVoltage > VOLTAGE_MAX) { faultMsg = "OVERVOLTAGE!";     faultActive = true; }
  if (busVoltage < VOLTAGE_MIN) { faultMsg = "UNDERVOLTAGE!";    faultActive = true; }
  if (current_A  > CURRENT_MAX) { faultMsg = "OVERCURRENT!";     faultActive = true; }
  if (temperature > TEMP_MAX)   { faultMsg = "OVERTEMP!";        faultActive = true; }

  if (faultActive) {
    digitalWrite(RELAY_PIN, LOW);    // Disconnect battery
    digitalWrite(BUZZER_PIN, HIGH);  // Sound alert
    Serial.print("⚠ FAULT: ");
    Serial.println(faultMsg);
  } else {
    digitalWrite(RELAY_PIN, HIGH);   // Keep battery connected
    digitalWrite(BUZZER_PIN, LOW);
  }
}

// ── Update OLED Display ───────────────────────────────────
void updateDisplay() {
  display.clearDisplay();
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);

  // Title
  display.setCursor(20, 0);
  display.setTextSize(1);
  display.println("== SMART BMS ==");
  display.drawLine(0, 10, 128, 10, SSD1306_WHITE);

  // Parameters
  display.setCursor(0, 14);
  display.print("Voltage : "); display.print(busVoltage, 2); display.println(" V");

  display.setCursor(0, 24);
  display.print("Current : "); display.print(current_A, 2);  display.println(" A");

  display.setCursor(0, 34);
  display.print("Temp    : "); display.print(temperature, 1); display.println(" C");

  display.setCursor(0, 44);
  display.print("SOC     : "); display.print(SOC, 0);         display.println(" %");

  display.setCursor(0, 54);
  if (faultActive) {
    display.setTextColor(SSD1306_BLACK, SSD1306_WHITE);
    display.print("  !! FAULT DETECTED !!  ");
  } else {
    display.print("SOH     : "); display.print(SOH, 0); display.print(" %");
  }

  display.display();
}

// ── Send data over CAN Bus ────────────────────────────────
void sendCANData() {
  // Pack voltage (x100 to send as integer)
  uint16_t v = (uint16_t)(busVoltage * 100);
  uint16_t i = (uint16_t)(current_A * 100);
  uint8_t  t = (uint8_t)(temperature + 40);  // offset -40 to +215
  uint8_t  s = (uint8_t)SOC;

  canMsg.can_id  = 0x100;
  canMsg.can_dlc = 6;
  canMsg.data[0] = (v >> 8) & 0xFF;
  canMsg.data[1] = v & 0xFF;
  canMsg.data[2] = (i >> 8) & 0xFF;
  canMsg.data[3] = i & 0xFF;
  canMsg.data[4] = t;
  canMsg.data[5] = s;
  mcp2515.sendMessage(&canMsg);
}

// ── SETUP ─────────────────────────────────────────────────
void setup() {
  Serial.begin(9600);
  Wire.begin();

  // OLED Init
  if (!display.begin(SSD1306_SWITCHCAPVCC, 0x3C)) {
    Serial.println("OLED not found!");
    while (true);
  }
  display.clearDisplay();
  display.setCursor(20, 25);
  display.setTextSize(1);
  display.setTextColor(SSD1306_WHITE);
  display.println("Initializing BMS...");
  display.display();
  delay(1500);

  // INA219 Init
  if (!ina219.begin()) {
    Serial.println("INA219 not found!");
    while (true);
  }
  ina219.setCalibration_32V_2A();

  // CAN Bus Init
  mcp2515.reset();
  mcp2515.setBitrate(CAN_500KBPS, MCP_8MHZ);
  mcp2515.setNormalMode();

  // Pins
  pinMode(RELAY_PIN,  OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, HIGH);   // Start with battery connected
  digitalWrite(BUZZER_PIN, LOW);

  lastTime = millis();
  Serial.println("BMS Ready!");
}

// ── LOOP ──────────────────────────────────────────────────
void loop() {
  unsigned long now = millis();
  float dt = (now - lastTime) / 3600000.0;  // Convert ms to hours
  lastTime = now;

  // Read sensors
  busVoltage  = ina219.getBusVoltage_V() + (ina219.getShuntVoltage_mV() / 1000.0);
  current_A   = ina219.getCurrent_mA() / 1000.0;
  temperature = readTemperature();

  // Coulomb counting for SOC
  chargeUsed_Ah += current_A * dt;
  SOC = ((BATTERY_CAPACITY_AH - chargeUsed_Ah) / BATTERY_CAPACITY_AH) * 100.0;
  SOC = constrain(SOC, 0.0, 100.0);

  // Estimate SOH (simplified — based on voltage at full charge)
  if (SOC > 95.0) SOH = (busVoltage / 50.4) * 100.0;
  SOH = constrain(SOH, 0.0, 100.0);

  // Check faults
  checkProtection();

  // Update display
  updateDisplay();

  // Serial monitor output
  Serial.print("V:"); Serial.print(busVoltage, 2);
  Serial.print(" I:"); Serial.print(current_A, 2);
  Serial.print(" T:"); Serial.print(temperature, 1);
  Serial.print(" SOC:"); Serial.print(SOC, 0);
  Serial.print("% SOH:"); Serial.print(SOH, 0);
  Serial.println("%");

  // Send CAN data every 500ms
  if (now - lastCanSend >= 500) {
    sendCANData();
    lastCanSend = now;
  }

  delay(1000);
}
